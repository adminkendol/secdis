200          private String login(String[] network) throws Throwable { 
201              //Basic config = new Basic(); 
202              //global variable 
203              Session session = new Session(getApplicationContext()); 
204   
205              String allDevice=config.getDeviceInfo("/"); 
206              Log.d("DEVICE INFO:",allDevice); 
207              String [] device =allDevice.split("/"); 
208              String token = session.getToken(); 
209              Log.d("FBase token", token); 
210   
211              String url = config.getLogin(); 
212              JSONParser jParser = new JSONParser(); 
213              String email = _emailText.getText().toString(); 
214              String password = _passwordText.getText().toString(); 
215              List<NameValuePair> parameter = new ArrayList<NameValuePair>(); 
216              parameter.add(new BasicNameValuePair("un", email)); 
217              parameter.add(new BasicNameValuePair("pass", password)); 
218              parameter.add(new BasicNameValuePair("token", token)); 
219              parameter.add(new BasicNameValuePair("device_id", device[12])); 
220              parameter.add(new BasicNameValuePair("brand", device[1])); 
221              parameter.add(new BasicNameValuePair("operator", network[2])); 
222              parameter.add(new BasicNameValuePair("mcc", network[3])); 
223              parameter.add(new BasicNameValuePair("mnc", network[4])); 
224              parameter.add(new BasicNameValuePair("lac", network[0])); 
225              parameter.add(new BasicNameValuePair("cid", network[1])); 
226              parameter.add(new BasicNameValuePair("address","")); 
227                      Log.d("URL SIGNIN", url); 
228              JSONObject json = jParser.makeHttpRequest(url, "POST", parameter); 
229              if(String.valueOf(json).equals("null")){ 
230                  return "Exception Caught"; 
231              }else { 
232                  Log.d("RESULT DATA", String.valueOf(json)); 
233                  return String.valueOf(json); 
234              } 
235          }
